```go
func lengthOfLongestSubstring(s string) int {
	charMap := make(map[uint8]int) //记录每个字母最近一次出现的下标
	start := 0 //记录目前观察无重复字符子串的起点（不一定是最长子串）
	maxLen := 0
	for i := 0; i < len(s); i++ {
		if start+maxLen > len(s) {
			break
		}
		if tem, ok := charMap[s[i]];!ok || tem < start {
			maxLen = max(maxLen, i-start+1)
		} else {
			start = tem + 1
		}
		charMap[s[i]] = i
	}
	return maxLen
}
```
